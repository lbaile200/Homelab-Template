CREATE TABLE csp_reports (     id INT AUTO_INCREMENT PRIMARY KEY,     log_date DATETIME,     ip_address VARCHAR(45),     forwarded_for TEXT,     status INT,     user_agent TEXT,     body_bytes_sent INT,     request TEXT,     blocked_uri TEXT,     violated_directive TEXT,     original_policy TEXT,     document_uri TEXT,     source_file TEXT,     line_number INT,     column_number INT,     script_sample TEXT,     created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP );
